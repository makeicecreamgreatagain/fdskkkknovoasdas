#property strict
#include <Trade/Trade.mqh>
CTrade trade;

//================ CONFIG =================//
input double RiskPercent          = 1.0;
input int    ATR_Period            = 14;
input double ATR_Min               = 0.0005;
input int    RSI_Period            = 14;
input int    StartHour             = 8;
input int    EndHour               = 18;
input double MaxSpread             = 20;
input double DailyLossLimitPercent = 2.0;

//================ HANDLES =================//
int handleEMA200;
int handleEMA20;
int handleRSI;
int handleATR;

//================ CONTROLE =================//
datetime lastTradeDay      = 0;
double   startOfDayBalance = 0;
datetime lastBarTime       = 0;

//+------------------------------------------------------------------+
//| OnInit: inicializa handles e estado                              |
//+------------------------------------------------------------------+
int OnInit()
{
   handleEMA200 = iMA(_Symbol, PERIOD_H1,  200,      0, MODE_EMA, PRICE_CLOSE);
   handleEMA20  = iMA(_Symbol, PERIOD_M15, 20,       0, MODE_EMA, PRICE_CLOSE);
   handleRSI    = iRSI(_Symbol, PERIOD_M15, RSI_Period, PRICE_CLOSE);
   handleATR    = iATR(_Symbol, PERIOD_M15, ATR_Period);

   if (handleEMA200 == INVALID_HANDLE ||
       handleEMA20  == INVALID_HANDLE ||
       handleRSI    == INVALID_HANDLE ||
       handleATR    == INVALID_HANDLE)
   {
      Print("Erro ao criar handles dos indicadores.");
      return INIT_FAILED;
   }

   startOfDayBalance = AccountInfoDouble(ACCOUNT_BALANCE);
   trade.SetComment("EA_Revisado");

   return INIT_SUCCEEDED;
}

//+------------------------------------------------------------------+
//| OnDeinit: libera handles                                         |
//+------------------------------------------------------------------+
void OnDeinit(const int reason)
{
   IndicatorRelease(handleEMA200);
   IndicatorRelease(handleEMA20);
   IndicatorRelease(handleRSI);
   IndicatorRelease(handleATR);
}

//================ FILTROS =================//

// Verifica se é um novo dia de trade e atualiza o balance de referência.
// Retorna true apenas na primeira chamada do dia.
// ATENÇÃO: só deve ser chamada quando todos os outros filtros já passaram,
// pois registra o dia como "disponível" e deve ser combinada com a flag
// de trade aberta para evitar múltiplas entradas.
bool IsNewTradeDay()
{
   MqlDateTime now, last;
   TimeToStruct(TimeCurrent(), now);
   TimeToStruct(lastTradeDay, last);

   if (now.day != last.day || now.mon != last.mon || now.year != last.year)
      return true;

   return false;
}

bool IsTradingTime()
{
   MqlDateTime dt;
   TimeToStruct(TimeCurrent(), dt);
   return (dt.hour >= StartHour && dt.hour <= EndHour);
}

bool IsWeekday()
{
   MqlDateTime dt;
   TimeToStruct(TimeCurrent(), dt);
   return (dt.day_of_week >= 1 && dt.day_of_week <= 5);
}

bool IsSpreadOK()
{
   double spread = (SymbolInfoDouble(_Symbol, SYMBOL_ASK) -
                    SymbolInfoDouble(_Symbol, SYMBOL_BID)) / _Point;
   return spread <= MaxSpread;
}

bool IsVolumeStrong()
{
   long vol = iVolume(_Symbol, PERIOD_M15, 0);
   long avg = 0;
   for (int i = 1; i <= 10; i++)
      avg += iVolume(_Symbol, PERIOD_M15, i);
   avg /= 10;
   return vol > avg;
}

// Limite de perda diária — compara equity atual com o balance
// registrado no início do dia (inclui trades já fechadas no dia).
bool HitDailyLossLimit()
{
   double equity = AccountInfoDouble(ACCOUNT_EQUITY);
   double loss   = (startOfDayBalance - equity) / startOfDayBalance * 100.0;
   return loss >= DailyLossLimitPercent;
}

// Detecta nova barra no M15 para evitar sinais duplicados no mesmo candle.
bool IsNewBar()
{
   datetime currentBar = iTime(_Symbol, PERIOD_M15, 0);
   if (currentBar != lastBarTime)
   {
      lastBarTime = currentBar;
      return true;
   }
   return false;
}

//================ CANDLE =================//

bool IsPinBar()
{
   double open  = iOpen (_Symbol, PERIOD_M15, 1);
   double close = iClose(_Symbol, PERIOD_M15, 1);
   double high  = iHigh (_Symbol, PERIOD_M15, 1);
   double low   = iLow  (_Symbol, PERIOD_M15, 1);

   double body      = MathAbs(close - open);
   double upperWick = high - MathMax(open, close);
   double lowerWick = MathMin(open, close) - low;

   return (upperWick > body * 2 || lowerWick > body * 2);
}

bool IsBullishEngulfing()
{
   double open1  = iOpen (_Symbol, PERIOD_M15, 1);
   double close1 = iClose(_Symbol, PERIOD_M15, 1);
   double open2  = iOpen (_Symbol, PERIOD_M15, 2);
   double close2 = iClose(_Symbol, PERIOD_M15, 2);

   return (close1 > open1 && close2 < open2 &&
           close1 > open2 && open1  < close2);
}

bool IsBearishEngulfing()
{
   double open1  = iOpen (_Symbol, PERIOD_M15, 1);
   double close1 = iClose(_Symbol, PERIOD_M15, 1);
   double open2  = iOpen (_Symbol, PERIOD_M15, 2);
   double close2 = iClose(_Symbol, PERIOD_M15, 2);

   return (close1 < open1 && close2 > open2 &&
           close1 < open2 && open1  > close2);
}

//================ LOTE =================//

double GetLotSize(double stopLossPoints)
{
   double balance  = AccountInfoDouble(ACCOUNT_BALANCE);
   double risk     = balance * (RiskPercent / 100.0);
   double tickVal  = SymbolInfoDouble(_Symbol, SYMBOL_TRADE_TICK_VALUE);

   if (tickVal <= 0 || stopLossPoints <= 0) return 0;

   double lot     = risk / (stopLossPoints * tickVal);
   double minLot  = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_MIN);
   double maxLot  = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_MAX);
   double lotStep = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_STEP);

   lot = MathMax(minLot, MathMin(maxLot, lot));
   lot = MathRound(lot / lotStep) * lotStep;

   return NormalizeDouble(lot, 2);
}

//================ BREAK EVEN =================//

void ApplyBreakEven()
{
   for (int i = 0; i < PositionsTotal(); i++)
   {
      ulong ticket = PositionGetTicket(i);
      if (!PositionSelectByTicket(ticket)) continue;

      // Filtra apenas posições do símbolo atual
      if (PositionGetString(POSITION_SYMBOL) != _Symbol) continue;

      double openPrice = PositionGetDouble(POSITION_PRICE_OPEN);
      double sl        = PositionGetDouble(POSITION_SL);
      double tp        = PositionGetDouble(POSITION_TP);   // preserva o TP original
      double price     = SymbolInfoDouble(_Symbol, SYMBOL_BID);
      double risk      = MathAbs(openPrice - sl);

      if (risk <= 0) continue;

      long posType = PositionGetInteger(POSITION_TYPE);

      if (posType == POSITION_TYPE_BUY && sl < openPrice)
      {
         if (price - openPrice >= risk)
            trade.PositionModify(ticket, openPrice, tp);
      }

      if (posType == POSITION_TYPE_SELL && sl > openPrice)
      {
         if (openPrice - price >= risk)
            trade.PositionModify(ticket, openPrice, tp);
      }
   }
}

//================ LEITURA DE INDICADORES =================//

bool GetIndicatorValues(double &ema200, double &ema20, double &rsi, double &atr)
{
   double bufEMA200[], bufEMA20[], bufRSI[], bufATR[];

   if (CopyBuffer(handleEMA200, 0, 0, 1, bufEMA200) <= 0) return false;
   if (CopyBuffer(handleEMA20,  0, 0, 1, bufEMA20)  <= 0) return false;
   if (CopyBuffer(handleRSI,    0, 0, 1, bufRSI)    <= 0) return false;
   if (CopyBuffer(handleATR,    0, 0, 1, bufATR)    <= 0) return false;

   ema200 = bufEMA200[0];
   ema20  = bufEMA20[0];
   rsi    = bufRSI[0];
   atr    = bufATR[0];

   return true;
}

//================ MAIN =================//

void OnTick()
{
   // Break-even roda em todo tick (não depende de nova barra)
   ApplyBreakEven();

   // Filtros de sessão
   if (!IsTradingTime())  return;
   if (!IsWeekday())      return;
   if (!IsSpreadOK())     return;
   if (!IsVolumeStrong()) return;
   if (HitDailyLossLimit()) return;

   // Só processa na abertura de nova barra
   if (!IsNewBar()) return;

   // Verifica se já operou hoje — sem efeito colateral aqui
   if (!IsNewTradeDay()) return;

   // Lê todos os indicadores de uma vez
   double ema200, ema20, rsi, atr;
   if (!GetIndicatorValues(ema200, ema20, rsi, atr)) return;

   if (atr < ATR_Min) return;

   double stopLossPoints  = atr * 2.0 / _Point;
   double takeProfitPoints = atr * 3.0 / _Point;

   double lot = GetLotSize(stopLossPoints);
   if (lot <= 0) return;

   // Usa o fechamento do candle anterior para os sinais
   double close1 = iClose(_Symbol, PERIOD_M15, 1);
   double price  = SymbolInfoDouble(_Symbol, SYMBOL_ASK);  // ASK para compra
   double bid    = SymbolInfoDouble(_Symbol, SYMBOL_BID);  // BID para venda

   bool buySignal  = (close1 > ema200) &&
                     (close1 <= ema20)  &&
                     (rsi < 40)         &&
                     (IsPinBar() || IsBullishEngulfing());

   bool sellSignal = (close1 < ema200) &&
                     (close1 >= ema20)  &&
                     (rsi > 60)         &&
                     (IsPinBar() || IsBearishEngulfing());

   //================ BUY =================//
   if (buySignal)
   {
      double sl = price - stopLossPoints  * _Point;
      double tp = price + takeProfitPoints * _Point;

      if (trade.Buy(lot, _Symbol, price, sl, tp))
         lastTradeDay = TimeCurrent();   // só marca o dia se a ordem foi enviada
   }

   //================ SELL =================//
   else if (sellSignal)
   {
      double sl = bid + stopLossPoints  * _Point;
      double tp = bid - takeProfitPoints * _Point;

      if (trade.Sell(lot, _Symbol, bid, sl, tp))
         lastTradeDay = TimeCurrent();   // só marca o dia se a ordem foi enviada
   }
}
